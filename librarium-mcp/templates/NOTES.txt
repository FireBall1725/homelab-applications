  kubectl -n {{ .Release.Namespace }} logs deploy/{{ include "common.names.fullname" . }}

Health check (no auth):
  kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "common.names.fullname" . }} 8090:8090
  curl http://localhost:8090/health
